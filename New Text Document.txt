str(Luban)
Luban$F <- factor(Luban$Color)
Luban$out <- relevel(Luban$F, ref="2")

mymodel <- multinom(out~ Age+Gender+edu+Height+Weight+Waist+BMI+Hip+WHR+SpO2+Bsys+Bdia+Drinks+Fastfood+BodyTemperature+PulseRate+Blooduricacid,data=Luban)
summary(mymodel)
predict(mymodel,Luban)
library(AER)
coeftest(mymodel)
exp(coef(mymodel))
